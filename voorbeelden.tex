\section{Combinatoriek}
\label{sec:combinatoriek}
\subsection*{Variaties}

\begin{vb}
  Het aantal variaties van $2$ elementen uit $4$ is $12$.
  \[ V_{4}^{2} = 4 \cdot 3 \]
\end{vb}

\begin{vb}
  Het aantal variaties van $3$ elementen uit $14$ is $2184$.
  \[ V_{14}^{3} = 4 \cdot 3 \]
\end{vb}
